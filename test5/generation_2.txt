import math

def compute_evaporation_time(mass):
    G = 6.6743e-11  # m^3 / (kg * s^2)
    hbar = 1.0546e-34  # J * s
    c = 2.9979e8  # m / s
    mass_kg = mass
    if mass > 1e+20:
        mass_kg = mass / 1e+20
    elif mass > 1e+10:
        mass_kg = mass / 1e+10
    elif mass > 1e+5:
        mass_kg = mass / 1e+5
    t = 5120 * math.pi * G**2 * mass_kg**3 / (hbar * c**4)
    return t
