def prime_sum():
    # Initialize a variable to store the sum
    prime_sum = 0
    # Initialize a counter to keep track of the number of prime numbers
    counter = 0
    # Initialize a number to test for prime
    num = 2
    # Loop until the counter is equal to 100
    while counter < 100:
        # Initialize a flag to check if the number is prime
        is_prime = True
        # Iterate from 2 to the number
        for i in range(2, num):
            # Check if the number is divisible by any of the numbers
            if (num % i == 0):
                # If divisible, set the flag to False
                is_prime = False
        # Check if the flag is still True
        if is_prime:
            # If True, add the number to the sum
            prime_sum += num
            # Increment the counter
            counter += 1
        # Increment the number
        num += 1
    # Return the sum
    return prime_sum
